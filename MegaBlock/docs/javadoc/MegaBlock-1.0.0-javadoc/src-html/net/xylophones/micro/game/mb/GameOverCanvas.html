013    package net.xylophones.micro.game.mb;
014    
015    import javax.microedition.lcdui.Image;
016    import javax.microedition.lcdui.Canvas;
017    import javax.microedition.lcdui.Graphics;
018    import javax.microedition.lcdui.Font;
019    
020    /**
021     * A simple class to display a page saying congrats for winning the game
022     *
023     * @author william@xylophones.net
024     */
025    public class GameOverCanvas extends Canvas {
026    
027        /**
028         * title text
029         */
030        private String title = "Game over";
031    
032        /**
033         * Left hand side of a bit of text telling the user to press a key to continue
034         * in the middle of this and continueTextRight will be the name of the button 
035         * they have to press to continue
036         */
037        private String continueTextLeft = "press ";
038        
039        /**
040         * Right hand side of a bit of text telling the user to press a key to continue
041         * in the middle of this and continueTextLeft will be the name of the button 
042         * they have to press to continue
043         */
044        private String continueTextRight = " to continue";
045    
046        /**
047         * Colour of the text
048         */
049        private int textColor = 0x0000FF;
050    
051        /**
052         * Background colour of the screen
053         */
054        int backgroundColor = 0x000000;
055        
056        /**
057         * The background image
058         */
059        private Image background = null;
060        
061        /**
062         * Padding around the text
063         */
064        private int textPadding = 10;
065        
066        /**
067         * Background colour of the text part 
068         */
069        private int textBackgroundColor = 0x000000;
070        
071        /**
072         * Width of the border around text block
073         */
074        private int textBorder = 2;
075        
076        /**
077         * Colour of the text
078         */
079        private int textBorderColor = 0xAA0000;
080        
081        /**
082         * Holds the Font object returned from the Graphics object - needed
083         * so that we can determine its height for spacing when painting on the 
084         * screen
085         */
086        private Font font;
087        
088        /**
089         * Used to switch between different screens - needed to display the 
090         * main menu when the user continues from this screen
091         */
092        private ScreenChooser screenChooser;
093    
094        /**
095         * Creates a new instance of WonCanvas
096         */
097        public GameOverCanvas(ScreenChooser screenChooser) {
098            setFullScreenMode(true);
099            this.screenChooser = screenChooser;
100        }
101    
102        /**
103         * <p>
104         * Paint the congrats to the screen. 
105         * </p>
106         *
107         * @param g
108         */
109        public void paint(Graphics g) {
110                    
111            // needed for dimentions
112            font = g.getFont();
113            
114            // background
115            if (background != null) {
116                g.drawImage(background, 0, 0, 0);
117            } else {
118                g.setColor(backgroundColor);
119                g.fillRect(0, 0, getWidth(), getHeight());
120            }
121            
122            // text background
123            int textBackgroundHeight = 2 * (textPadding + font.getHeight());
124            int textBackgroundYPos = getHeight() / 2 - font.getHeight() - textPadding;
125            
126            g.setColor(textBackgroundColor);
127            g.fillRect(0, textBackgroundYPos, getWidth(), textBackgroundHeight);
128            
129            g.setColor(textBorderColor);
130            g.fillRect(0, textBackgroundYPos-textBorder, getWidth(), textBorder);
131            g.fillRect(0, textBackgroundHeight+textBackgroundYPos, getWidth(), textBorder);
132    
133            // text
134            g.setColor(textColor);
135    
136            int yPos = getHeight() / 2 - font.getHeight();
137            int xPos = getWidth() / 2;
138            g.drawString(title, xPos, yPos, Graphics.TOP|Graphics.HCENTER);
139    
140            String continueText = continueTextLeft
141                                  + getKeyName( getKeyCode(Canvas.FIRE) ).toUpperCase()
142                                  + continueTextRight;
143            
144            yPos = getHeight() / 2;
145            xPos = getWidth() / 2;
146            g.drawString(continueText, xPos, yPos, Graphics.TOP|Graphics.HCENTER);
147        }
148        
149        /**
150         * Listen for the continue keypress - the {@code Canvas.FIRE} keypress -
151         * and when it is received, display the main menu screen
152         *
153         * @param keyCode
154         */
155        public void keyPressed(int keyCode) {
156            if (getGameAction(keyCode) == Canvas.FIRE) {
157                try {
158                    screenChooser.displayScreen(ScreenChooser.SCREEN_MAIN_MENU);
159                } catch (Exception e) {
160                }
161            }
162        }
163        
164        /**
165         * Set the background image
166         * 
167         * @param background
168         */
169        public void setBackground(Image background) {
170            this.background = background;
171        }
172    }
