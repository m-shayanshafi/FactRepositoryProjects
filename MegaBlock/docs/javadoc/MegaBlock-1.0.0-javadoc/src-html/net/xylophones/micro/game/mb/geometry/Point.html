014    package net.xylophones.micro.game.mb.geometry;
015    
016    /**
017     * A simple class consisting of an x,y set of coordinates
018     */
019    public class Point {
020    
021            private int x;
022            
023            private int y;
024            
025            public int getX() {
026                    return x;
027            }
028    
029            public void setX(int x) {
030                    this.x = x;
031            }
032    
033            public int getY() {
034                    return y;
035            }
036    
037            public void setY(int y) {
038                    this.y = y;
039            }
040    
041            public Point(int x, int y) {
042                    this.x = x;
043                    this.y = y;
044            }
045            
046    }
