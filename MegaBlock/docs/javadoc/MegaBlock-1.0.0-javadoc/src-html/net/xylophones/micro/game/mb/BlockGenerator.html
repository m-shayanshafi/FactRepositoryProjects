013    package net.xylophones.micro.game.mb;
014    
015    import java.util.Random;
016    
017    /**
018     * Class for generating a random tetrad
019     * 
020     * @todo - this shouldn't be completely random
021     *
022     * @author william@xylophones.net
023     */
024    public class BlockGenerator {
025        
026            /**
027             * Used for generating rendom numbers
028             */
029        Random random;
030        
031        /**
032         * Integer array holding all possible tetrad 
033         * identifiers - one of the {@code TETRAD_X} constants 
034         * in the {@code Block} class 
035         */
036        int[] possibleTetrads = Block.getPossibleTetrads();
037    
038        /**
039         * Creates a new instance of BlockGenerator
040         */
041        public BlockGenerator() {
042            random = new Random( System.currentTimeMillis() );
043        }
044        
045        /**
046         * Get a random tetrad
047         * 
048         * Returns an integer identifier which corresponds to
049         * one of the {@code TETRAD_X} constants in the {@code Block} class 
050         *
051         * @return identifier
052         * 
053         * @see net.xylophones.micro.game.mb.Block
054         */
055        public int randomTetrad() {
056            int tetrad = random.nextInt(possibleTetrads.length);
057            
058            return possibleTetrads[tetrad];
059        }
060        
061    }
