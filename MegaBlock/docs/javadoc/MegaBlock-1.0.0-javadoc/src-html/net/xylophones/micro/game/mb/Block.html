017    /**
018     * <p>
019     * A class which defines the shapes of each tetrad in each
020     * possible rotation and provides methods for determining certain 
021     * characteristics of each shape. 
022     * </p>
023     * 
024     * <p>
025     * Each possible tetrad has a constant value 
026     * {@code TETRAD_I, TETRAD_J, TETRAD_L} etc. 
027     * </p>
028     * 
029     * <p>
030     * Each possible tetrad rotation constant value 
031     * - one of {@code ROTATION_0, ROTATION_90, ROTATION_180} or {@code ROTATION_270}
032     * </p>  
033     * 
034     * <p>
035     * Combined, a tertad and a rotation have a <i>shape</i>. A shape
036     * is a 4x4 array of booleans, where {@code true} values represent a filled
037     * space and a {@code false} values repeesent a space. 
038     * </p>  
039     * 
040     * <p>
041     * e.g. The J tetrad in the {@code ROTATION_0} position <br />
042     * <tt>
043     * tfff<br />
044     * tttf<br />
045     * ffff<br />
046     * ffff
047     * </tt>
048     * </p>
049     * 
050     *
051     * @author william@xylophones.net
052     */

335        /**
336         * Get the shape of a tetrad based on the tetrad identifier
337         * and a rotation
338         */
339        public static boolean[][] getShape(int tetrad, int rotation) {
340            return tetradShapes[rotation][tetrad];
341        }
342    
343        /**
344         * <p>
345         * Get the leftmost cells of the block.
346         * </p>
347         * 
348         * <p>
349         * Returns an array of {@code Point} objects holding the
350         * position of each filled cell in the first column from
351         * the left which has a filled cell.
352         * </p>
353         * 
354         * <p>
355         * The coordinates in the {@code Point} objects are the
356         * location in the tetrad shape
357         * </p>
358         * 
359         * @param tetrad
360         * @param rotation
361         *
362         * @return 
363         */
364        public static Point[] firstLeftCells(int tetrad, int rotation) {
365            boolean[][] shape = getShape(tetrad, rotation);
366    
367            Point[] tmpPoints = new Point[4];
368    
369            int numFound = 0;
370    
371            for (int y=3 ; y>=0 ; y--) {
372                for (int x=0 ; x<=3 ; x++) {
373                    if (shape[x][y]) {
374                        tmpPoints[numFound] = new Point(x, y);
375                        numFound++;
376                        break;
377                    }
378                }
379            }
380            
381            Point[] points = new Point[numFound];
382            System.arraycopy(tmpPoints, 0, points, 0, numFound);
383            
384            return points;
385        }
